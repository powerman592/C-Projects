#include <iostream>
using namespace std;

class Dog {
public:
	int age;
	double weight;
	string name;
	string furColor;
	string breed;

	Dog(int dogAge, double dogWeight, string dogName, string dogColor, string dogBreed)
	{
		age = dogAge;
		weight = dogWeight;
		name = dogName;
		furColor = dogColor;
		breed = dogBreed;
	}

	void bark();
	string rename(string newName);
	double eat(double foodAmount);
};

void Dog::bark() {
	cout << "Woof! Woof!";
}
double Dog::eat(double amountOfFOod) {
	weight += amountOfFOod;
	return weight;
}
string Dog::rename(string theNewName) {
	name = theNewName;
	return name;
}

int main()
{
	int dogAge;
	double dogWeight, foodAmount;
	string dogName, dogColor, dogBreed, newName;

	cout << "You are about to create a dog.";
	cout << "How old is the dog: ";
	cin >> dogAge;
	cout << "How much does the dog weigh: ";
	cin >> dogWeight;
	cout << "What is the dog\'s name: ";
	cin >> dogName;
	cout << "What color is the dog: ";
	cin >> dogColor;
	cout << "What breed is the dog: ";
	cin >> dogBreed;

	Dog dog1(dogAge, dogWeight, dogName, dogColor, dogBreed);

	cout << endl;
	cout << dog1.name << " is a " << dog1.age << " year old " << dog1.furColor 
		<< " " << dog1.breed << " that weighs " << dog1.weight << " lbs.";
	cout << endl << endl;

	dog1.bark();

	cout << endl << "Patches is hungry, how much should he eat: ";
	cin >> foodAmount;
	dog1.eat(foodAmount);

	cout << "Patches isn\'t a very good name. What should they be renamed to: ";
	cin >> newName;
	dog1.rename(newName);

	cout << endl;
	cout << dog1.name << " is a " << dog1.age << " year old " << dog1.furColor
		<< " " << dog1.breed << " that weighs " << dog1.weight << " lbs.";


	
	return 0;
}
